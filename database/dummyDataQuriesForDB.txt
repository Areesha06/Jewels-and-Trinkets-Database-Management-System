PRAGMA foreign_keys = ON;

-- ============================
-- USERS
-- ============================
CREATE TABLE Users (
    user_id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    email TEXT UNIQUE NOT NULL,
    phone TEXT,
    password TEXT NOT NULL,
    role TEXT CHECK (role IN ('admin', 'staff', 'customer')) NOT NULL,
    created_at TEXT DEFAULT (datetime('now'))
);

-- ============================
-- CATEGORY
-- ============================
CREATE TABLE Category (
    cat_id INTEGER PRIMARY KEY AUTOINCREMENT,
    categoryName TEXT NOT NULL
);

-- ============================
-- ITEMS
-- ============================
CREATE TABLE Items (
    item_id INTEGER PRIMARY KEY AUTOINCREMENT,
    itemName TEXT NOT NULL,
    basePrice REAL NOT NULL,
    image BLOB
);

-- ============================
-- ITEMSQUANT
-- ============================
CREATE TABLE ItemsQuant (
    itemQuant_id INTEGER PRIMARY KEY AUTOINCREMENT,
    item_id INTEGER NOT NULL,
    color TEXT,
    stock INTEGER,
    description TEXT,
    FOREIGN KEY (item_id) REFERENCES Items(item_id)
);

-- ============================
-- SUBCATEGORY
-- ============================
CREATE TABLE SubCategory (
    subcat_id INTEGER PRIMARY KEY AUTOINCREMENT,
    cat_id INTEGER NOT NULL,
    item_id INTEGER NOT NULL,
    subCatName TEXT,
    FOREIGN KEY (cat_id) REFERENCES Category(cat_id),
    FOREIGN KEY (item_id) REFERENCES Items(item_id)
);

-- ============================
-- ADDRESS
-- ============================
CREATE TABLE Address (
    addr_id INTEGER PRIMARY KEY AUTOINCREMENT,
    province TEXT,
    city TEXT,
    area TEXT,
    houseNumber TEXT
);

-- ============================
-- USERADDRESS
-- ============================
CREATE TABLE UserAddress (
    user_id INTEGER NOT NULL,
    addr_id INTEGER NOT NULL,
    PRIMARY KEY (user_id, addr_id),
    FOREIGN KEY (user_id) REFERENCES Users(user_id),
    FOREIGN KEY (addr_id) REFERENCES Address(addr_id)
);

-- ============================
-- ORDERS
-- ============================
CREATE TABLE Orders (
    order_id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    staff_id INTEGER,
    addr_id INTEGER NOT NULL,
    totalPrice REAL,
    paymentMethod TEXT,
    order_status TEXT,
    order_date TEXT DEFAULT (datetime('now')),
    dispatch_date TEXT,
    trackingID TEXT,
    FOREIGN KEY (user_id) REFERENCES Users(user_id),
    FOREIGN KEY (staff_id) REFERENCES Users(user_id),
    FOREIGN KEY (addr_id) REFERENCES Address(addr_id)
);

-- ============================
-- ORDERITEMS
-- ============================
CREATE TABLE OrderItems (
    orderItem_id INTEGER PRIMARY KEY AUTOINCREMENT,
    order_id INTEGER NOT NULL,
    itemQuant_id INTEGER NOT NULL,
    quantity INTEGER,
    price REAL,
    FOREIGN KEY (order_id) REFERENCES Orders(order_id),
    FOREIGN KEY (itemQuant_id) REFERENCES ItemsQuant(itemQuant_id)
);

-- ============================
-- CART
-- ============================
CREATE TABLE Cart (
    cart_id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    itemQuant_id INTEGER NOT NULL,
    quantity INTEGER,
    FOREIGN KEY (user_id) REFERENCES Users(user_id),
    FOREIGN KEY (itemQuant_id) REFERENCES ItemsQuant(itemQuant_id)
);

-- ============================
-- WISHLIST
-- ============================
CREATE TABLE Wishlist (
    wishlist_id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    itemQuant_id INTEGER NOT NULL,
    FOREIGN KEY (user_id) REFERENCES Users(user_id),
    FOREIGN KEY (itemQuant_id) REFERENCES ItemsQuant(itemQuant_id)
);

-- ============================
-- COMPLAINT
-- ============================
CREATE TABLE Complaint (
    complaint_id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    description TEXT,
    status TEXT,
    created_at TEXT DEFAULT (datetime('now')),
    FOREIGN KEY (user_id) REFERENCES Users(user_id)
);
